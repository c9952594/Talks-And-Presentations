var add_task = isNaN($routeParams.task_id)
if (add_task) {
    $scope.task = {};
    $scope.editing = false;
} else {
    taskFactory.getTask($routeParams.task_id).success(function (task) {
        $scope.task = task;
    });
    $scope.editing = true;
}

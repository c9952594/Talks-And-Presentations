angular.module('demo', ['ngRoute']);

---------------

angular.module('demo').config(function ($routeProvider) {
    $routeProvider
        .when('/tasks', {
            controller: 'TasksController',
            templateUrl: 'tasks.html'
        })
        .otherwise({
            redirectTo: '/tasks'
        });
});
